/**
 * Button module
 * @module @krupnik/button
 * @see module:my/shirt
 * @author Yuri Krupnik [krupnik.yuri@gmail.com]
 * @return array
 */


const Button = (props) => {
    const { children, onClick } = props; // eslint-disable-line
    return (
        <button
            onClick={onClick}
            className={styles.root}
            type="button"
        >
            <div>
                {children}
            </div>
        </button>
    );
};

Button.defaultProps = {
    children: null
};

Button.propTypes = {
    children: PropTypes.oneOfType([
        PropTypes.node,
    ])
};
export default React.memo(Button);
